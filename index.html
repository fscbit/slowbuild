# 一个用户上传的脚本大概长这样：

# batch-rename-dates.sh
# 批量给照片文件名加上拍摄日期

for f in *.jpg; do
  date=$(exiftool -d "%Y%m%d" -DateTimeOriginal "$f" | cut -d' ' -f2)
  mv "$f" "${date}_${f}"
done

echo "✓ 搞定"
